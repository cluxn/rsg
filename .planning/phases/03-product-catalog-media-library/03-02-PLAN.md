# Plan 03-02: Frontend Product Pages (SSG + ISR)

## Objective

Build the 10 public-facing product category pages in Next.js using SSG with ISR. Each page renders a full-width hero (real photo or stock + gradient overlay), two-column layout (constrained image + content with sticky CTA), and a full-width specs table below the fold. The "Get Quote" CTA is rendered as a stub (Phase 4 wires submission logic).

## must_haves

```yaml
truths:
  - /products/polycarbonate-sheet renders with full-width hero, two-column section (image left, content right), specs table, and orange "Get Quote" CTA button
  - /products/colour-coated-roofing-sheet renders without a blank hero (has stock photo + gradient-premium overlay)
  - generateStaticParams returns all 10 product slugs (verified by running next build without errors for all 10 routes)
  - Product pages use revalidate = 3600 for ISR fallback
  - Primary product image is constrained to max-w-[500px] — not stretched full-width
  - Specs table is hidden (not rendered) when product.specs is null or empty
  - Get Quote CTA button renders on every product page with id="get-quote-section" anchor
  - All product images use next/image with explicit alt text from media.alt_text
  - Mobile layout is single column: hero → primary image → title → description → CTA → specs
```

```typescript
const BACKEND_URL = process.env.BACKEND_URL || 'http://localhost:3001';

export type ProductMedia = { id: number; url: string; alt_text: string; display_order: number };
export type Product = {
  id: number; slug: string; name: string;
  description: string | null;
  specs: { label: string; value: string }[] | null;
  media: ProductMedia[];
};

export async function getProducts(): Promise<{ slug: string; name: string }[]> {
  const res = await fetch(`${BACKEND_URL}/api/products`, { next: { tags: ['products'] } });
  if (!res.ok) throw new Error('Failed to fetch products');
  return res.json();
}

export async function getProduct(slug: string): Promise<Product | null> {
  const res = await fetch(`${BACKEND_URL}/api/products/${slug}`, {
    next: { tags: [`product-${slug}`] },
  });
  if (res.status === 404) return null;
  if (!res.ok) throw new Error('Failed to fetch product');
  return res.json();
}
```

```tsx
import { notFound } from 'next/navigation';
import Image from 'next/image';
import { getProduct, getProducts } from '@/lib/api';
import { ProductHero } from '@/components/ui/ProductHero';
import { SpecsTable } from '@/components/ui/SpecsTable';
import { GetQuoteCTA } from '@/components/ui/GetQuoteCTA';
import { CTAButton } from '@/components/ui/CTAButton';

export const revalidate = 3600;

export async function generateStaticParams() {
  const products = await getProducts();
  return products.map(p => ({ slug: p.slug }));
}

export async function generateMetadata({ params }: { params: { slug: string } }) {
  const product = await getProduct(params.slug);
  if (!product) return {};
  return {
    title: `${product.name} | RSG Profile Manufacturing`,
    description: product.description?.slice(0, 155) ?? `${product.name} - RSG Profile Manufacturing Pvt. Ltd.`,
  };
}

// Map slug to hero image path (all images in /public/products/)
const heroImageMap: Record<string, string> = {
  'colour-coated-roofing-sheet': '/products/colour-coated-roofing-sheet-hero.jpg',
  'ms-plate-channel-angle': '/products/ms-plate-channel-angle-hero.jpg',
  'ms-pipe': '/products/ms-pipe-hero.jpg',
  'decking-sheet': '/products/decking-sheet-hero.jpg',
  'purlins': '/products/purlins-hero.jpg',
  'polycarbonate-sheet': '/products/polycarbonate-sheet-hero.jpg',
  'crimping-sheet': '/products/crimping-sheet-hero.jpg',
  'self-drilling-screws': '/products/self-drilling-screws-hero.jpg',
  'turbo-air-ventilator': '/products/turbo-air-ventilator-hero.jpg',
  'accessories': '/products/accessories-hero.jpg',
};

export default async function ProductPage({ params }: { params: { slug: string } }) {
  const product = await getProduct(params.slug);
  if (!product) notFound();

  const heroImage = heroImageMap[product.slug] ?? '/products/accessories-hero.jpg';
  const primaryMedia = product.media[0] ?? null;
  const additionalMedia = product.media.slice(1);

  return (
    <>
      <ProductHero
        productName={product.name}
        heroImagePath={heroImage}
        tagline="RSG Profile Manufacturing"
      />

      {/* Two-column section */}
      <section className="max-w-[1280px] mx-auto px-6 lg:px-16 py-12 lg:py-16">
        <div className="grid grid-cols-1 lg:grid-cols-2 gap-8 lg:gap-12">
          {/* Left: images */}
          <div>
            {primaryMedia ? (
              <div className="max-w-[500px]">
                <Image
                  src={primaryMedia.url}
                  alt={primaryMedia.alt_text}
                  width={500}
                  height={400}
                  className="rounded-xl shadow-md w-full h-auto object-cover"
                />
              </div>
            ) : (
              <div className="max-w-[500px] rounded-xl bg-steel/10 aspect-[5/4] flex items-center justify-center">
                <span className="font-body text-navy/30 text-sm">No image yet</span>
              </div>
            )}
            {additionalMedia.length > 0 && (
              <div className="flex gap-3 mt-4 flex-wrap">
                {additionalMedia.map(m => (
                  <Image
                    key={m.id}
                    src={m.url}
                    alt={m.alt_text}
                    width={80}
                    height={80}
                    className="rounded-lg object-cover border border-navy/10 hidden lg:block"
                  />
                ))}
              </div>
            )}
          </div>

          {/* Right: content + sticky CTA */}
          <div className="lg:relative">
            <div className="lg:sticky lg:top-24">
              <h2 className="font-heading text-navy text-2xl lg:text-3xl font-bold mb-4">
                {product.name}
              </h2>
              {product.description && (
                <p className="font-body text-navy/80 text-base lg:text-[17px] leading-relaxed mb-6">
                  {product.description}
                </p>
              )}
              {/* Key spec highlights (first 3 specs) */}
              {product.specs && product.specs.length > 0 && (
                <dl className="space-y-2 mb-8">
                  {product.specs.slice(0, 3).map((s, i) => (
                    <div key={i} className="flex gap-2 text-sm">
                      <dt className="font-heading text-navy font-semibold min-w-[140px]">{s.label}:</dt>
                      <dd className="font-body text-navy/70">{s.value}</dd>
                    </div>
                  ))}
                </dl>
              )}
              <CTAButton href="#get-quote-section" className="w-full lg:w-auto px-8 py-3 text-base">
                Get Quote
              </CTAButton>
            </div>
          </div>
        </div>
      </section>

      {/* Full-width specs table */}
      <SpecsTable specs={product.specs} />

      {/* Get Quote CTA band */}
      <GetQuoteCTA productName={product.name} />
    </>
  );
}
```
