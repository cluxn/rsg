2. **`listLeads` handler** (new export):
   ```ts
   export async function listLeads(req: Request, res: Response): Promise<void> {
     const page = Math.max(1, parseInt(String(req.query.page || '1'), 10));
     const limit = Math.min(100, Math.max(1, parseInt(String(req.query.limit || '25'), 10)));
     try {
       const { leads, total } = await getLeads(page, limit);
       res.json({ leads, total, page, limit });
     } catch (err) {
       console.error('listLeads error:', err);
       res.status(500).json({ error: 'Failed to fetch leads' });
     }
   }
   ```

3. **`exportLeads` handler** (new export):
   ```ts
   export async function exportLeads(req: Request, res: Response): Promise<void> {
     try {
       const csv = await exportLeadsToCSV();
       const date = new Date().toISOString().slice(0, 10);
       res.setHeader('Content-Type', 'text/csv');
       res.setHeader('Content-Disposition', `attachment; filename="rsg-leads-${date}.csv"`);
       res.send(csv);
     } catch (err) {
       console.error('exportLeads error:', err);
       res.status(500).json({ error: 'Export failed' });
     }
   }
   ```

     if (!req.file) { res.status(400).json({ error: 'No CSV file uploaded' }); return; }
     try {
       const result = await importLeadsFromCSV(req.file.buffer);
       res.json(result);
     } catch (err) {
       console.error('importLeads error:', err);
       res.status(500).json({ error: 'Import failed' });
     }
   }

<action>
1. **Update `backend/src/routes/leads.ts`**:
   ```ts
   import { Router } from 'express';
   import multer from 'multer';
   import { submitLead, listLeads, exportLeads, importLeads } from '../controllers/leads.controller';
   import { requireAuth } from '../middleware/auth';

   const upload = multer({ storage: multer.memoryStorage(), limits: { fileSize: 5 * 1024 * 1024 } });
   const router = Router();

   router.post('/', submitLead);                                    // public — POST /api/leads
   router.get('/', requireAuth, listLeads);                         // admin — GET /api/leads
   router.get('/export', requireAuth, exportLeads);                 // admin — GET /api/leads/export
   router.post('/import', requireAuth, upload.single('file'), importLeads); // admin — POST /api/leads/import

   export default router;
   ```
   Note: use an inline `multer` instance here (memory storage, 5MB limit) rather than importing the media upload middleware (which has different limits/filters for images).

2. **Update `backend/src/index.ts`** — add rate limiter for the leads submission route:
   ```ts
   import rateLimit from 'express-rate-limit';
   // ...existing imports...

   const leadsRateLimit = rateLimit({
     windowMs: 15 * 60 * 1000, // 15 minutes
     max: 10,
     standardHeaders: true,
     legacyHeaders: false,
     message: { error: 'Too many enquiries from this IP. Please try again later.' },
   });

   // Apply before the leadsRouter mount:
   app.use('/api/leads', leadsRateLimit);
   app.use('/api/leads', leadsRouter);
   ```
   Place the `leadsRateLimit` middleware line immediately before the existing `app.use('/api/leads', leadsRouter)` line (rate limit applies to all `/api/leads` routes, but at 10 req/15min the admin export/import won't be affected in practice; the guard is for public spam submissions).
