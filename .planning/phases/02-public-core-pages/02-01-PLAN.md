```ts
const API_BASE = process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:4000';

export async function getSettings(): Promise<Record<string, string>> {
  const res = await fetch(`${API_BASE}/api/settings`, {
    next: { revalidate: 3600 },
  });
  if (!res.ok) throw new Error(`getSettings failed: ${res.status}`);
  return res.json();
}
```

This layout:
1. Fetches settings via `getSettings()` at render time. Wrap in try/catch — if the backend is unreachable, use an empty fallback object (the WhatsApp button will be hidden, footer will still render).
2. Extracts `whatsapp_number` from settings (empty string if key is missing).
3. Renders: `SiteHeader` → `<main>{children}</main>` → `SiteFooter` → `WhatsAppFloat`

```tsx
import { SiteHeader } from '@/components/layout/SiteHeader';
import { SiteFooter } from '@/components/layout/SiteFooter';
import { WhatsAppFloat } from '@/components/layout/WhatsAppFloat';
import { getSettings } from '@/lib/api';

export default async function SiteLayout({ children }: { children: React.ReactNode }) {
  let settings: Record<string, string> = {};
  try {
    settings = await getSettings();
  } catch {
    // backend unreachable — WhatsApp button hidden, other layout elements still render
  }
  return (
    <>
      <SiteHeader />
      <main>{children}</main>
      <SiteFooter />
      <WhatsAppFloat number={settings.whatsapp_number ?? ''} />
    </>
  );
}
```
